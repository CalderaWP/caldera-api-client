/**
 * Standard bas class for an API client
 *
 * Based on CF Pro v1 API Client
 */
export class Client {
	route:string;
	/**
	 * Request Mode
	 */
	mode:string;
	headers:Headers;
	credentials:string;
	constructor( route:string, headers: ?Object = {} ){
		this.route = route;
		this.mode = 'same-origin';
		this.headers = new Headers(headers);
		this.credentials = 'same-origin';
	}

	/**
	 * Make a request
	 * @param endpoint Endpoint of route to request
	 * @param data
	 * @param method
	 * @returns {Request}
	 */
	request(endpoint:string,data:options,method:string) {
		let request = this.createRequest(endpoint,data,method);
		return fetch(request);
	}

	createRequest(endpoint:string,data:options,method:string) :Request {
		let args = {
			method: 'method',
			mode: this.mode,
			credentials: this.credentials,
			redirect: 'follow',
		};

		args.headers = new Headers();
		if ('POST' === method || 'PUT' === method) {
			args.body = JSON.stringify(data);
			args.headers.append("Content-Type", "application/json");
			args.headers.append("Content-Length", args.body.length.toString());
		}

		return new Request(this.urlFromEndpoint(endpoint, method), args);

	}

	urlFromEndpoint(endpoint:string,method:string = 'GET', data:?object = {}) : string {
		if( 'GET'=== method){
			return this.urlString(data);
		}

		return `${this.route}${endpoint}`
	}

	reqGet(data:object,endpoint:string=''): object {
		return this.request(endpoint,data,'GET')
			.then(
				response => {return this.handleResponse(response);},
				error => {return this.handleError(error);}
			);
	}

	reqPost(data:object) {
		return this.request(endpoint,data,'POST')
			.then(
				response => {return this.handleResponse(response);},
				error => {return this.handleError(error);}
			);
	}

	reqPut(data:object) {
		return this.request(endpoint,data,'PUT')
			.then(
				response => {return this.handleResponse(response);},
				error => {return this.handleError(error);}
			);
	}

	reqDelete() {
		return this.request(endpoint,{},'DELETE')
			.then(
				response => {return this.handleResponse(response);},
				error => {return this.handleError(error);}
			);

	}

	handleResponse(response: Response ) : object{
		if ('string' === typeof response.body) {
			const maybe = JSON.parse(response.body);
			if ('object' === typeof  maybe) {
				return maybe;
			}

		}
		return response.body;
	}

	handleError(error) {
		if (true === this.doAlerts) {

			const 	message = error.response.data.message;

			if( 402 == error.response.status ){
				return;
			}else{
				alert(message);
			}
		}
		return error;
	}

	urlString(data:object,endpoint:string=''):string {
		let str = "";
		for (let key in data) {
			if (str != "") {
				str += "&";
			}
			str += key + "=" + data[key];
		}
		if( endpoint){
			return `${this.route}/${endpoint}?${str}`
		}
		return this.route + '?' + str;
	}

}
